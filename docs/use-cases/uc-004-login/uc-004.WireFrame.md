```mermaid
%% Wireframe for Use Case 004: User Login
classDiagram
  direction TB

  namespace UI {
    class LoginPage {
      +UsernameInput: string
      +PasswordInput: string
      +RememberMe: bool
      +LoginButton()
      +ForgotPasswordLink()
      +NotificationArea: string
    }
  }

  namespace Core.DTOs.Account {
    class LoginRequestDto {
      +Username: string
      +Password: string
    }

    class LoginResponseDto {
      +JwtToken: string "nullable"
      +RefreshToken: string "nullable"
      +ErrorMessage: string "nullable"
    }
  }

  namespace Core.Services {
    class AccountService {
      +LoginAsync(username: string, password: string): LoginResult
    }

    class NotificationHandler {
      +Info(message: string)
      +Error(message: string)
    }
  }

  namespace Infrastructure.Identity {
    class AccountManager {
      +AuthenticateAsync(username: string, password: string): AuthResult
      +GenerateJwtToken(user: object): string
    }
  }

  namespace WebApi.Controllers {
    class LoginApiController {
      +Login(CredentialsDto): LoginResult
    }
  }

  %% Associations
  LoginPage --> LoginRequestDto : creates
  LoginPage --> LoginApiController : submits credentials
  LoginApiController --> AccountService : uses
  AccountService --> LoginRequestDto : receives
  AccountService --> AccountManager : authenticates
  AccountService --> NotificationHandler : notifies
  AccountManager --> LoginResponseDto : returns
  LoginResponseDto --> LoginPage : returns result
  NotificationHandler --> LoginPage : displays

  %% Notes
  %% - LoginPage contains the visible login wireframe elements.
  %% - UsernameInput and PasswordInput map to LoginRequestDto fields.
  %% - LoginButton submits credentials to LoginApiController.
  %% - AccountService handles login logic.
  %% - AccountManager authenticates the user and generates tokens.
  %% - NotificationHandler displays success or error messages to the UI.
```

---

## Notes
- UsernameInput and PasswordInput collect the login credentials.
- LoginButton submits the credentials for authentication.
- LoginRequestDto transfers username and password.
- LoginResponseDto returns tokens or an error message.
- NotificationHandler displays login success or failure feedback.
